Only High People

people_80 <- DF %>% 
  group_by(observation, class) %>% 
  summarize(correct = mean(as.numeric(correct), na.rm = T)) %>% 
  filter(correct >= .80) %>% 
  ungroup() %>% 
  group_by(observation) %>% 
  summarize(two = n()) %>% 
  filter(two == 2) %>% 
  pull(observation)
## `summarise()` has grouped output by 'observation'. You can override using the
